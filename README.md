# c_sven
Aprender palabras en ingles jugando un juego de deletriar letra por letra

# Requerimientos
    libgtk-4-1
    questk
    
# Install questk
    git clone https://github.com/rsvzz/simple_str_list.git
    cd src/model
    meson build
    ninja -C build
    cd build
    sudo ninja install

# Install
    meson build
    meson compile -C build
    cd build
    sudo ninja install
    
# Uninstall
    sudo ninja uninstall
